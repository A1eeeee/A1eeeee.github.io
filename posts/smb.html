smb

  1. 安装smb
sudo apt install samba
  1. 设置特定用户访问
#设置用户名
sudo touch /etc/samba/用户名称
#设置密码
sudo smbpasswd -a 密码
  1. 修改配置文件
#进入配置文件
sudo vim /etc/samba/smb.conf

# 最后添加以下内容

[share]
comment = 填写介绍内容
path = 填写文件路径
writable = yes
valid user = 天锡克smb用户名
  1. 端口
#开放445端口
ufw allow 445
部署Linux服务
